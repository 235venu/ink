use super::log_info;

/// Extracts the Wasm blob from a contract bundle.
pub fn extract_wasm(contract: &contract_metadata::ContractMetadata) -> Vec<u8> {
    let code = contract
        .source
        .wasm
        .clone()
        .expect("contract bundle is missing `source.wasm`");
    log_info(&format!(
        "{:?} has {} KiB",
        contract.contract.name,
        code.0.len() / 1024
    ));
    code.0
}
